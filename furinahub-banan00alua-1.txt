-- LOAD UI
local Library = loadstring(game:HttpGet("https://raw.githubusercontent.com/FluxusTeam/fluxlib/main/library.lua"))()

-- WINDOW (THÊM TIÊU ĐỀ)
local Window = Library:Window(" furina Hub | by 1phut36giay")

-- LOGO (ICON GÓC MÀN HÌNH)
pcall(function()
    Library:Notification("Banana Hub Loaded ", "Welcome!", 5)
end)

-- TẠO IMAGE LOGO TRONG GAME
local ScreenGui = Instance.new("ScreenGui", game.CoreGui)
local Logo = Instance.new("ImageLabel", ScreenGui)

Logo.Size = UDim2.new(0, 100, 0, 100)
Logo.Position = UDim2.new(0, 10, 0, 10)
Logo.BackgroundTransparency = 1
Logo.Image = "rbxassetid://104286684857067" -- bạn có thể đổi ID logo

-- TAB
local Main = Window:Tab("Main")
local Farm = Window:Tab("Auto Farm")
local Teleport = Window:Tab("Teleport")
local Player = Window:Tab("Player")

-- VARIABLES
_G.AutoFarm = false
_G.AutoClick = false
_G.AutoStats = false

-- MAIN
Main:Button("Check Level", function()
    print("Level:", game.Players.LocalPlayer.Data.Level.Value)
end)

Main:Toggle("Auto Click", false, function(v)
    _G.AutoClick = v
end)

-- AUTO FARM
Farm:Toggle("Auto Farm", false, function(v)
    _G.AutoFarm = v
end)

spawn(function()
    while wait() do
        if _G.AutoFarm then
            pcall(function()
                local player = game.Players.LocalPlayer
                local enemy = workspace.Enemies:FindFirstChildOfClass("Model")
                if enemy and enemy:FindFirstChild("HumanoidRootPart") then
                    player.Character.HumanoidRootPart.CFrame =
                        enemy.HumanoidRootPart.CFrame * CFrame.new(0,0,3)
                end
            end)
        end
    end
end)

-- AUTO CLICK
spawn(function()
    while wait() do
        if _G.AutoClick then
            game:GetService("VirtualUser"):Button1Down(Vector2.new(0,0))
            game:GetService("VirtualUser"):Button1Up(Vector2.new(0,0))
        end
    end
end)

-- TELEPORT
Teleport:Button("Teleport Island", function()
    game.Players.LocalPlayer.Character.HumanoidRootPart.CFrame =
        CFrame.new(0, 50, 0)
end)

-- PLAYER
Player:Slider("WalkSpeed", 16, 100, 16, function(v)
    game.Players.LocalPlayer.Character.Humanoid.WalkSpeed = v
end)

Player:Slider("JumpPower", 50, 200, 50, function(v)
    game.Players.LocalPlayer.Character.Humanoid.JumpPower = v
end)

-- AUTO STATS
Player:Toggle("Auto Stats Melee", false, function(v)
    _G.AutoStats = v
end)

spawn(function()
    while wait() do
        if _G.AutoStats then
            game:GetService("ReplicatedStorage").Remotes.CommF_:InvokeServer("AddPoint","Melee",1)
        end
    end
end)
